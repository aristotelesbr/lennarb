def initialize(name)
  @name = name.to_sym

  return if NAMES.include?(@name)

  raise ArgumentError, "Invalid environment: #{@name.inspect}"
end

def on(*envs)
  matched = envs.include?(:any) ||
    envs.include?(:all) ||
    envs.include?(name) ||
    (envs.include?(:local) && local?)

  yield if matched
end
