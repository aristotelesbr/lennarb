def match_and_execute_route(req, res)
  params     = {}
  path_parts = split_path(req.path_info)
  endpoint   = find_endpoint(@root_node, path_parts, params)

  if endpoint && (action = endpoint[req.request_method])
    req.params.merge!(params)
    action.call(req, res)
  else
    res.not_found
  end
end
